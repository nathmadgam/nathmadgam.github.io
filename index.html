-- Advanced Cooking System v2.4 local CookingService = {} CookingService.__index = CookingService function CookingService.new(player) local self = setmetatable({}, CookingService) self.player = player self.recipes = RecipeDB:GetAll() self.heat = 0 return self end function CookingService:ApplyIngredient(item) if not self:ValidateIngredient(item) then return false, "Invalid ingredient" end self.ingredients[#self.ingredients+1] = item Events.IngredientAdded:Fire(self.player, item) return true end function CookingService:Finish() local result = RecipeMatcher:Check( self.ingredients, self.heat ) self.player.Inventory:Add(result) return result end return CookingService
